\begin{enumerate}
    \item QaTa-COV19: bao gồm 9258 ảnh X-ray ngực được ghi chú thủ công thương tổn do COVID-19 lần đầu tiên. Nhóm nghiên cứu đã mở rộng ghi chú với sự hỗ trợ của các chuyên gia. Ví dụ: "Unilateral pulmonary infection, two infected areas, middle lower left lung" (tạm dịch: "Một bên phổi bị nhiễm bệnh, hai vùng bị nhiễm, ở giữa phía dưới phổi trái").
    \item ESO-CT: bao gồm 286 ảnh X-ray được ghi chú độc lập bởi hai chuyên gia.

\begin{frame}{\secname: \subsecname: CNN}
    Nhánh CNN nhận đầu vào là một ảnh và đầu ra là mặt nạ phân đoạn. 
    \begin{columns}
        \column{0.5\linewidth}
            \begin{align*}
                D_i &= DownCNN_i = CTBN \\
                    &= ReLU(BatchNorm_i(Conv_i(\cdot)))
            \end{align*}
            \[Y_{DownCNN, i+1} = MaxPool(D_i(Y_{DownCNN, i}))\]
            \[Y_{DownCNN, 1} = I\]

        \column{0.5\linewidth}
            \hspace{0.3\linewidth}
            \includegraphics[width=0.6\linewidth]{images/cnn-branch.png}
    \end{columns}
\end{frame}

\begin{frame}{\secname: \subsecname: ViT}
    Nhánh ViT nhận đầu vào là đặc trưng ảnh của khối DownCNN đầu tiên cùng với đặc trưng văn bản từ lớp BERT-Embed từ BERT\_12\_768\_12.
    \begin{columns}
        \column{0.4\linewidth}
            \includegraphics[width=\linewidth]{images/vit-branch.png}
        
        \column{0.6\linewidth}
            \[ViT(x) = ViT_2(ViT_1(x))\]
            \[x' = ViT_1(x) = MHSA(LN(x)) + x\]
            \[Y = ViT_2(x') = MLP(LN(x')) + x'\]
            \[x_{img, i} = PatchEmbedding(Y_{DownCNN, i})\]
            \[Y_{DownViT, 1} = ViT(x_{img, 1} + CTBN(x_{text}))\]
            \[Y_{DownViT, i+1} = ViT(Y_{DownViT, i} + x_{img, i + 1})\]

    \end{columns}
\end{frame}
